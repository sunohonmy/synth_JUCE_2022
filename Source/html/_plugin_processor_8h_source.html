19class WavetableOscillator
20{
21public:
22 WavetableOscillator (juce::AudioSampleBuffer& wavetableToUse)
23 : wavetable (wavetableToUse),
24 tableSize (wavetable.getNumSamples() - 1)
25 {
26 jassert (wavetable.getNumChannels() == 1);
27 }
28
29 void setFrequency (float frequency, float sampleRate)
30 {
31 auto tableSizeOverSampleRate = (float) tableSize / sampleRate;
32 tableDelta = frequency * tableSizeOverSampleRate;
33 }
34
35 forcedinline float getNextSample() noexcept
36 {
37
38 auto index0 = (unsigned int) currentIndex;
39 auto index1 = index0 + 1;
40
41 auto frac = currentIndex - (float) index0;
42
43 auto* table = wavetable.getReadPointer (0);
44 auto value0 = table[index0];
45 auto value1 = table[index1];
46
47 auto currentSample = value0 + frac * (value1 - value0);
48
49 if ((currentIndex += tableDelta) > (float) tableSize)
50 currentIndex -= (float) tableSize;
51
52 return currentSample;
53 }
54
55 void setWavetable(const juce::AudioSampleBuffer& wavetableToUse)
56 {
57 jassert (wavetableToUse.getNumChannels() == 1);
58
59 wavetable = wavetableToUse;
60 tableSize = wavetable.getNumSamples() - 1;
61
62 }
63
64 juce::AudioSampleBuffer& getWavetable()
65 {
66 return wavetable;
67 }
68
69private:
70 juce::AudioSampleBuffer& wavetable;
71 int tableSize;
72 float currentIndex = 0.0f, tableDelta = 0.0f;
73};

160 void createSineWavetable()
161 {
162 sineTable.setSize (1, (int) tableSize + 1);
163 sineTable.clear();
164
165 auto* samples = sineTable.getWritePointer (0);
166
167 auto angleDelta = juce::MathConstants<double>::twoPi / (double) (tableSize);
168 auto currentAngle = 0.0;
169 for(unsigned int i = 0; i < tableSize; ++i)
170 {
171 auto sample = std::sin(currentAngle);
172 samples[i] += (float) sample;
173 currentAngle += angleDelta;
174 }
175 samples[tableSize] = samples[0];
176
177
178 }

180 void createSawWavetable()
181 {
182 sawTable.setSize(1, (int)tableSize + 1);
183 sawTable.clear();
184
185 auto* samples = sawTable.getWritePointer(0);
186
187 auto angleDelta = 1.0 / (double)(tableSize);
188 auto currentAngle = 0.0;
189 for (unsigned int i = 0; i < tableSize; ++i)
190 {
191 auto sample = 2.0 * (currentAngle - floor(0.5 + currentAngle));
192 samples[i] += (float)sample;
193 currentAngle += angleDelta;
194 }
195 samples[tableSize] = samples[0];
196
197
198 }

200 void createSquareWavetable()
201 {
202 squareTable.setSize(1, (int)tableSize + 1);
203 squareTable.clear();
204
205 auto* samples = squareTable.getWritePointer(0);
206
207 auto angleDelta = 1.0 / (double)(tableSize);
208 auto currentAngle = 0.0;
209 for (unsigned int i = 0; i < tableSize; ++i)
210 {
211 auto sample = 2.0 * (2.0 * floor(currentAngle) - floor(2.0 * currentAngle)) + 1.0;
212 samples[i] += (float)sample;
213 currentAngle += angleDelta;
214 }
215 samples[tableSize] = samples[0];

226
227 auto angleDelta = 1.0 / (double)(tableSize);
228 auto currentAngle = 0.0;
229 for (unsigned int i = 0; i < tableSize; ++i)
230 {
231 auto sample = 4.0 * abs(currentAngle - floor(currentAngle + 0.75) + 0.25) - 1.0;
232 samples[i] += (float)sample;
233 currentAngle += angleDelta;
234 }
